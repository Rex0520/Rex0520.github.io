SpringBoot学习总结#

使用SpringBoot的优缺点#

优点：#

  1. 简化配置
  2. 整合了很多工具，如redis
  3. 开发效率高

缺点：#

  1. 因为内部封装了太多的东西，不容易找bug
  2. 不容易看到底层都做了什么

SpringBoot常用注解#

Controller:#

​ @Controller:通知Spring这个类为Controller类

​ @ResponseBody：用于将函数的返回值转换为json格式的数据写入到http响应的body区中,若没有这个注解，会使用请求转发跳转到返回值的页面

​ @RestController：Rest风格的Controller，@Controller + @ResponseBody = @RestController

​ @ControllerAdvice：环绕增强的Controller，一般用于全局异常处理、全局数据绑定、全局数据预处理

​ @RequestMapping()：请求地址映射，一般用于描述url的前缀地址

​ @PostMapping()：post请求的地址映射

​ @GetMaping()：get请求的地址映射

​ @Valid：用于校验参数，需要编写全局异常捕捉类

​ @RequestBody：接收来自requestBody中的参数，即请求体中的参数

​ @RequestParm：有三个参数，required表示是否必须，默认为true，defaultValue设置参数默认值，value为接受url的参数名

​ @PathVariable：获取路径变量

​ @Size：限制传入参数的最大值(max)和最小值(min)，指定异常的message信息，可被@Valid校验检测到

​ @NotNull：限制传入的参数非空，指定异常的message信息，可被@Valid校验检测到

​ @Max:指定最大值value，指定异常的message信息，可被@Valid校验检测到

​ @Min：指定最小值value，指定异常的message信息，可被@Valid校验检测到

Swagger:#

​ @Api：用在类上，说明该类的作用。可以标记一个Controller类做为swagger 文档资源

​ @ApiOperation：用在方法上，说明方法的作用，每一个url资源的定义

​ @ApiParam：用在变量上，说明请求属性的作用

​ @ApiResponse：响应配置信息

​ @ApiReponses：响应集配置

​ @ReponseHeader：响应头设置

Service:#

​ @Service：用在类上，说明该类为Service类

Dao:#

​ @Repository：用在类上，说明该类为Dao类

Config:#

​ @Configuration：用在类上，说明该类为配置类

AOP：#

​ @Aspect：用在类上，将当前类标识为一个切面供容器读取

​ @Pointcut：用在方法上，描述方法的触发范围

​ @Before：用在方法上，在方法执行前执行

​ @AfterReturing：用在方法上，方法正常退出时执行

事务：#

​ @Transactional：开启事务

Redis：#

​ @Cacheable：value参数表示redis中存储的key值

其他：#

​ @Autowired：注入对象

<Configuration status="fatal">

    <Property name="baseDir" value="G:\var\logs\mall"/>

  <Appenders>
    <Console name="Console" target="SYSTEM_OUT">
      <!--控制台只输出level及以上级别的信息（onMatch），其他的直接拒绝（onMismatch） -->
      <ThresholdFilter level="info" onMatch="ACCEPT"
        onMismatch="DENY"/>
      <PatternLayout
        pattern="[%d{MM:dd HH:mm:ss.SSS}] [%level] [%logger{36}] - %msg%n"/>
    </Console>

    <!--debug级别日志文件输出-->
    <RollingFile name="debug_appender" fileName="${baseDir}/debug.log"
      filePattern="${baseDir}/debug_%i.log.%d{yyyy-MM-dd}">
      <!-- 过滤器 -->
      <Filters>
        <!-- 限制日志级别在debug及以上在info以下 -->
        <ThresholdFilter level="debug"/>
        <ThresholdFilter level="info" onMatch="DENY" onMismatch="NEUTRAL"/>
      </Filters>
      <!-- 日志格式 -->
      <PatternLayout pattern="[%d{HH:mm:ss:SSS}] [%p] - %l - %m%n"/>
      <!-- 策略 -->
      <Policies>
        <!-- 每隔一天转存 -->
        <TimeBasedTriggeringPolicy interval="1" modulate="true"/>
        <!-- 文件大小 -->
        <SizeBasedTriggeringPolicy size="100 MB"/>
      </Policies>
    </RollingFile>

    <!-- info级别日志文件输出 -->
    <RollingFile name="info_appender" fileName="${baseDir}/info.log"
      filePattern="${baseDir}/info_%i.log.%d{yyyy-MM-dd}">
      <!-- 过滤器 -->
      <Filters>
        <!-- 限制日志级别在info及以上在error以下 -->
        <ThresholdFilter level="info"/>

package com.imooc.mall.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import springfox.documentation.builders.ApiInfoBuilder;
import springfox.documentation.builders.PathSelectors;
import springfox.documentation.builders.RequestHandlerSelectors;
import springfox.documentation.service.ApiInfo;
import springfox.documentation.spi.DocumentationType;
import springfox.documentation.spring.web.plugins.Docket;

@Configuration
public class SpringFoxConfig {

    //访问http://localhost:8083/swagger-ui.html可以看到API文档
    @Bean
    public Docket api() {
        return new Docket(DocumentationType.SWAGGER_2)
                .apiInfo(apiInfo())
                .select()
                .apis(RequestHandlerSelectors.any())
                .paths(PathSelectors.any())
                .build();
    }

    private ApiInfo apiInfo() {
        return new ApiInfoBuilder()
                .title("慕慕生鲜")
                .description("")
                .termsOfServiceUrl("")
                .build();
    }
}

package com.imooc.mall.config;

import org.springframework.context.annotation.Configuration;
import org.springframework.web.servlet.config.annotation.ResourceHandlerRegistration;
import org.springframework.web.servlet.config.annotation.ResourceHandlerRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

/**
 * 配置地址映射
 * @author Rex
 * @create 2021-02-09 13:22
 */
@Configuration
public class ImoocMallWebMvcConfig implements WebMvcConfigurer {
    @Override
    public void addResourceHandlers(ResourceHandlerRegistry registry) {
        registry.addResourceHandler("swagger-ui.html").addResourceLocations(
                "classpath:/META-INF/resources/");
        registry.addResourceHandler("/webjars/**").addResourceLocations(
                "classpath:/META-INF/resources/webjars/");

    }
}

    package com.imooc.mall.config;

import java.time.Duration;
import org.springframework.cache.annotation.EnableCaching;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.redis.cache.RedisCacheConfiguration;
import org.springframework.data.redis.cache.RedisCacheManager;
import org.springframework.data.redis.cache.RedisCacheWriter;
import org.springframework.data.redis.connection.RedisConnectionFactory;

@Configuration
@EnableCaching
public class CachingConfig {

    @Bean
    public RedisCacheManager redisCacheManager(RedisConnectionFactory connectionFactory) {

        RedisCacheWriter redisCacheWriter = RedisCacheWriter.lockingRedisCacheWriter(connectionFactory);
        RedisCacheConfiguration cacheConfiguration = RedisCacheConfiguration.defaultCacheConfig();
        cacheConfiguration = cacheConfiguration.entryTtl(Duration.ofSeconds(30));

        RedisCacheManager redisCacheManager = new RedisCacheManager(redisCacheWriter, cacheConfiguration);
        return redisCacheManager;
    }
}

  3. 在application.properties中配置redis的host、port、password

  4. 在启动类上加上@EnableCaching注解，表示开启缓存

  5. 在要加上缓存的方法中加入注解@Cacheable(value = “name”)
